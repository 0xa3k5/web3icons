# Next.js Integration

Complete guide for using @web3icons/react with Next.js 13+ App Router and Pages Router.

## Installation

<InstallTabs pkg="@web3icons/react" />

## App Router (Recommended)

### Server Components

Individual icon components work directly in Server Components:

```tsx
// app/page.tsx (Server Component)
import { TokenBTC, TokenETH, NetworkEthereum } from '@web3icons/react'

export default function Page() {
  return (
    <div className="flex gap-4">
      <TokenBTC size={32} variant="branded" />
      <TokenETH size={32} variant="branded" />
      <NetworkEthereum size={32} variant="branded" />
    </div>
  )
}
```

### Client Components

Dynamic components require the `'use client'` directive:

```tsx
// app/components/token-display.tsx
'use client'

import { TokenIcon } from '@web3icons/react/dynamic'

interface Props {
  symbol: string
}

export function TokenDisplay({ symbol }: Props) {
  return <TokenIcon symbol={symbol} size={32} variant="branded" />
}
```

```tsx
// app/page.tsx
import { TokenDisplay } from './components/token-display'

export default function Page() {
  return <TokenDisplay symbol="BTC" />
}
```

### Hybrid Approach

Combine server and client components for optimal performance:

```tsx
// app/portfolio/page.tsx
import { TokenBTC, TokenETH } from '@web3icons/react'
import { TokenList } from './token-list'

// Static icons render on the server
// Dynamic list is a client component
export default function Portfolio() {
  return (
    <div>
      <header className="flex gap-2">
        <TokenBTC size={24} variant="branded" />
        <TokenETH size={24} variant="branded" />
      </header>
      <TokenList />
    </div>
  )
}
```

```tsx
// app/portfolio/token-list.tsx
'use client'

import { TokenIcon } from '@web3icons/react/dynamic'
import { useUserTokens } from '@/hooks/use-user-tokens'

export function TokenList() {
  const tokens = useUserTokens()

  return (
    <ul>
      {tokens.map((token) => (
        <li key={token.symbol} className="flex items-center gap-2">
          <TokenIcon symbol={token.symbol} size={24} variant="branded" />
          <span>{token.name}</span>
        </li>
      ))}
    </ul>
  )
}
```

## Pages Router

All components work in Pages Router without special configuration:

```tsx
// pages/index.tsx
import { TokenBTC, TokenETH } from '@web3icons/react'
import { TokenIcon } from '@web3icons/react/dynamic'

export default function Home() {
  return (
    <div>
      {/* Static imports */}
      <TokenBTC size={32} variant="branded" />
      <TokenETH size={32} variant="branded" />

      {/* Dynamic import */}
      <TokenIcon symbol="USDC" size={32} variant="branded" />
    </div>
  )
}
```

## Code Splitting

### Lazy Load Dynamic Icons

Split the dynamic icon bundle from your main bundle:

```tsx
'use client'

import { lazy, Suspense } from 'react'

const TokenIcon = lazy(() =>
  import('@web3icons/react/dynamic').then((mod) => ({
    default: mod.TokenIcon,
  })),
)

function IconSkeleton() {
  return <div className="h-8 w-8 animate-pulse rounded-full bg-gray-200" />
}

export function LazyTokenIcon({ symbol }: { symbol: string }) {
  return (
    <Suspense fallback={<IconSkeleton />}>
      <TokenIcon symbol={symbol} size={32} variant="branded" />
    </Suspense>
  )
}
```

### Route-Based Splitting

Dynamic icons in client components are automatically code-split per route:

```tsx
// app/wallet/page.tsx
'use client'

import { TokenIcon } from '@web3icons/react/dynamic'

// This chunk only loads when /wallet is visited
export default function WalletPage() {
  const assets = useWalletAssets()
  return assets.map((a) => <TokenIcon key={a.symbol} symbol={a.symbol} />)
}
```

### Optimal Import Patterns

```tsx
// ✅ - tree-shakeable, minimal bundle
import { TokenBTC, TokenETH } from '@web3icons/react'

// ✅ - only load when needed

// ❌ - imports entire icon library
import * as Icons from '@web3icons/react'
```

## SSR Considerations

Individual icon components are fully SSR-compatible and render on the server.

Dynamic components are client-only and will:

1. Render nothing during SSR
2. Hydrate on the client
3. Display the icon after hydration

Handle the hydration gap with fallbacks:

```tsx
'use client'

import { TokenIcon } from '@web3icons/react/dynamic'

export function Token({ symbol }: { symbol: string }) {
  return (
    <TokenIcon
      symbol={symbol}
      size={32}
      variant="branded"
      fallback={<div className="h-8 w-8 rounded-full bg-gray-100" />}
    />
  )
}
```

## Image Optimization

For using icons as images with Next.js Image component:

```tsx
import Image from 'next/image'
import { svgs } from '@web3icons/core'

export function TokenImage({ symbol }: { symbol: string }) {
  const key = `branded${symbol.toUpperCase()}` as keyof typeof svgs.tokens
  const src = svgs.tokens[key]

  return <Image src={src} alt={symbol} width={32} height={32} />
}
```

## TypeScript Configuration

No additional TypeScript configuration is needed. The packages include full type definitions.

```typescript
// Types are automatically inferred
import { TokenBTC } from '@web3icons/react'
import type { ComponentProps } from 'react'

type TokenBTCProps = ComponentProps<typeof TokenBTC>
// { size?: number | string; variant?: 'mono' | 'branded' | 'background'; ... }
```

## Common Patterns

### Token Display Component

```tsx
'use client'

import { TokenIcon } from '@web3icons/react/dynamic'

interface TokenDisplayProps {
  symbol: string
  name: string
  balance: string
}

export function TokenDisplay({ symbol, name, balance }: TokenDisplayProps) {
  return (
    <div className="flex items-center gap-3 rounded-lg border p-4">
      <TokenIcon symbol={symbol} size={40} variant="branded" />
      <div>
        <div className="font-semibold">{name}</div>
        <div className="text-sm text-gray-500">
          {balance} {symbol}
        </div>
      </div>
    </div>
  )
}
```

### Network Selector

```tsx
'use client'

import { NetworkIcon } from '@web3icons/react/dynamic'

const networks = [
  { chainId: 1, name: 'Ethereum' },
  { chainId: 137, name: 'Polygon' },
  { chainId: 42161, name: 'Arbitrum' },
]

export function NetworkSelector({
  onSelect,
}: {
  onSelect: (chainId: number) => void
}) {
  return (
    <div className="flex gap-2">
      {networks.map((network) => (
        <button
          key={network.chainId}
          onClick={() => onSelect(network.chainId)}
          className="flex items-center gap-2 rounded-lg border p-2 hover:bg-gray-50"
        >
          <NetworkIcon chainId={network.chainId} size={24} variant="branded" />
          <span>{network.name}</span>
        </button>
      ))}
    </div>
  )
}
```
