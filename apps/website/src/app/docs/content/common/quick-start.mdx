# @web3icons/common

Shared TypeScript types and metadata for all Web3 Icons packages. Use this package to access icon metadata, or build type-safe integrations.

## Installation

<InstallTabs pkg="@web3icons/common" />

> This package is included as a dependency of `@web3icons/react` and `@web3icons/core`. Install it directly only if you need standalone access to metadata or types.

## Metadata Access

Access comprehensive metadata for all icons:

```typescript
import {
  tokens,
  networks,
  wallets,
  exchanges,
} from '@web3icons/common/metadata'

// Find by symbol
const bitcoin = tokens.find((t) => t.symbol === 'BTC')
// { id: 'bitcoin', name: 'Bitcoin', symbol: 'BTC', ... }

// Find by chain ID
const ethereum = networks.find((n) => n.chainId === 1)
// { id: 'ethereum', name: 'Ethereum', chainId: 1, ... }
```

## Metadata Interfaces

`TTokenMetadata`

```typescript
interface TTokenMetadata {
  id: string // "bitcoin"
  name: string // "Bitcoin"
  symbol: string // "BTC"
  variants: TVariant[] // ["mono", "branded", "background"]
  addresses?: Record<string, string> // Contract addresses by network
}
```

`TNetworkMetadata`

```typescript
interface TNetworkMetadata {
  id: string // "ethereum"
  name: string // "Ethereum"
  shortName?: string // "eth"
  chainId?: number // 1
  variants: TVariant[] // ["mono", "branded"]
  nativeCoinId?: string // "ethereum"
}
```

`TWalletMetadata`

```typescript
interface TWalletMetadata {
```

## Search and Filter

### Search Tokens

```typescript
function searchTokens(query: string) {
  const q = query.toLowerCase()
  return tokens.filter(
    (token) =>
      token.name.toLowerCase().includes(q) ||
      token.symbol.toLowerCase().includes(q),
  )
}

// Usage
const results = searchTokens('bitcoin')
// [{ id: 'bitcoin', name: 'Bitcoin', symbol: 'BTC', ... }, ...]
```

### Find Network by Chain ID

```typescript
function getNetworkByChainId(chainId: number) {
  return networks.find((network) => network.chainId === chainId)
}

// Usage
const ethereum = getNetworkByChainId(1)
const polygon = getNetworkByChainId(137)
const arbitrum = getNetworkByChainId(42161)
```

### Find Token by Contract Address

```typescript
function getTokenByAddress(address: string, network: string) {
  return tokens.find(
    (token) =>
      token.addresses?.[network]?.toLowerCase() === address.toLowerCase(),
  )
}

// Usage
const usdc = getTokenByAddress(
  '0xa0b86991c6218b36c1d19d4a2e9eb0ce3606eb48',
  'ethereum',
)
```

## Type-Safe Components

Build components with full type safety:

```typescript

interface TokenCardProps {
  token: TTokenMetadata
  variant: TVariant
  onClick: (token: TTokenMetadata) => void
}

function TokenCard({ token, variant, onClick }: TokenCardProps) {
  return (
    <button onClick={() => onClick(token)}>
      <span>{token.name} ({token.symbol})</span>
    </button>
  )
}
```
